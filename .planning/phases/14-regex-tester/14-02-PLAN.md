      export function buildRegex(source, flags): { re: RegExp } | { error: string }
        - const enumFlags = flags.includes("g") ? flags : flags + "g";  // matchAll needs g (Pitfall 3)
        - try return { re: new RegExp(source, enumFlags) }
        - catch return { error: (err as Error).message }  // RGX-07 native message verbatim (UI-SPEC)

      export function enumerate(text, re): RegexMatch[]
        - return [...text.matchAll(re)].map((m) => ({
            index: m.index ?? 0, length: m[0].length, full: m[0],
            groups: m.slice(1),                      // numbered (RGX-02)
            named: m.groups ? { ...m.groups } : {},  // named (RGX-02)
          }))

      export function applyReplace(text, re, repl): string
        - return text.replace(re, repl);  // native $1/$<name>/$& — DO NOT hand-roll (Don't Hand-Roll)

      export function runRegex(req): RegexResult
        - if (req.source === "" || req.text === "") return { empty: true };  // D-13
        - const built = buildRegex(req.source, req.flags);
        - if ("error" in built) return { error: built.error };               // RGX-07
        - const matches = enumerate(req.text, built.re);
        - if (req.replace === undefined) return { matches };
        - const userRe = new RegExp(req.source, req.flags);  // USER's true flags (g all-vs-first, D-07)
        - return { matches, replaced: applyReplace(req.text, userRe, req.replace) };
