export interface BuildLatestJsonInput {
  version: string;     // app semver, e.g. "0.2.1"
  pubDate: string;     // RFC3339 timestamp, INJECTED by caller (D-03 — never Date.now here)

export function buildLatestJson(input: BuildLatestJsonInput): LatestJson;

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-05 | Tampering | buildLatestJson signature freshness | accept | Freshness (single-match glob of THIS build's `.sig`, fail loud on 0/>1) is the CALLER'S responsibility in Phase 11 (REL-06), by design (D-03 keeps this core pure/injectable). Documented as a caller contract; not enforceable in a pure assembler. |
| T-09-06 | Spoofing | dual-key url/signature mismatch | mitigate | The dual-key helper derives both arch entries from the SAME single `{ url, signature }` input, so the two keys cannot diverge or carry a swapped signature. | T-09-07 | Tampering | no invented darwin-universal key | mitigate | Output emits exactly the two real arch keys the updater queries; unit test asserts `not.toHaveProperty("darwin-universal")` so a stray key (which the updater would ignore, risking a silent no-update) can't slip in. |
