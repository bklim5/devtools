    ```ts
    import { useEffect } from "react";
    import { usePreferences } from "./usePreferences";
    import { useEntitlements } from "./useEntitlements";
    import { gatePreferences } from "@/lib/entitlements/entitlements";
    import { applyAppearance } from "./theme";

    export const THEME_HINT_KEY = "td-theme-hint";

    /** D-23-9/D-23-5: the ONE place whole-app appearance is applied. Applies the
     *  GATED effective theme+accent (free/downgraded → dark + #5b9bf8, never raw
     *  prefs — Pitfall 3) and keeps the localStorage paint-hint in sync from that
     *  gated value (Pitfall 4 — the pre-paint script reads it next launch). */
    export function useAppearance(): void {
      const { preferences, prefsLoaded } = usePreferences();
      const ents = useEntitlements();

      useEffect(() => {
        if (!prefsLoaded) return; // the pre-paint script set the launch theme
        const eff = gatePreferences(preferences, ents);
        applyAppearance(eff.theme, eff.accent);
        try { localStorage.setItem(THEME_HINT_KEY, eff.theme); } catch { /* never block on storage */ }
      }, [preferences.theme, preferences.accent, ents, prefsLoaded, preferences]);

      // Live OS light↔dark flip — subscribe ONLY while the effective theme is system.
      useEffect(() => {
        if (!prefsLoaded) return;
        const eff = gatePreferences(preferences, ents);
        if (eff.theme !== "system") return;
        const mq = window.matchMedia("(prefers-color-scheme: dark)");
        const onChange = () => applyAppearance("system", eff.accent);
        mq.addEventListener("change", onChange);
        return () => mq.removeEventListener("change", onChange);
      }, [preferences.theme, preferences.accent, ents, prefsLoaded, preferences]);
    }
    ```

    ```html
      // No-flash theme (D-23-5). The authoritative theme lives in the async Tauri
      // store (unreadable at first paint — MEMORY: tauri-store-async-init-race), so
      // we read a synchronous localStorage paint-hint written from the GATED value

          if (light) document.documentElement.dataset.theme = "light";
        } catch (e) { /* never block paint on a storage error */ }
      })();
    </script>
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| localStorage paint-hint → pre-paint script | The hint is non-authoritative, user-readable/writable synchronous storage consumed by the first-paint script. |
| entitlement set → whole-app apply | A free/downgraded user must be forced to default appearance regardless of persisted Pro values. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-23-03 | Elevation of Privilege | useAppearance / App root | mitigate | The App root applies `gatePreferences(preferences, ents)`, never raw prefs (Pitfall 3) — and the paint-hint is written from the GATED theme, so neither the live app nor the launch frame can render a free user's persisted Pro theme. The apply effect depends on `ents`, so a live downgrade re-forces dark + #5b9bf8. Client-side UX gate by design (v1.6 model — accepted, not a server secret). |
| T-23-06 | Tampering | localStorage td-theme-hint | accept | A user hand-editing the hint to "light" only affects the FIRST paint frame; the App-root effect immediately re-applies the gated authoritative value (correcting drift). The hint can never grant a Pro theme persistently — gatePreferences re-forces defaults for a free user on every apply. Cosmetic, single-frame, self-correcting; low risk, documented. |
| T-23-07 | Denial of Service | pre-paint script | mitigate | The script is wrapped in try/catch and never throws (a storage error is swallowed) so it can never block first paint (RESEARCH Pattern 4). Synchronous + tiny; no network, no async. |
</threat_model>
